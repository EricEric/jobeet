#Install Composer
curl -sS https://getcomposer.org/installer | php
mv composer.phar /usr/local/bin/composer

#Create symfony project
composer create-project symfony/framework-standard-edition path/to/install 2.4.*

#Go to config
localhost:port/config.php

#List console command
php app/console list

#Create bundle
php app/console generate:bundle --namespace=Ibw/JobeetBundle

#DB create
php app/console doctrine:database:create

#DB update schema
php app/console doctrine:schema:update --force

#DB load fixtures
php app/console doctrine:fixtures:load

#CRUD
php app/console doctrine:generate:crud --entity=IbwJobeetBundle:Job --route-prefix=ibw_job --with-write --format=yml

#Gen web asset
php app/console assets:install web --symlink

#Route Debugging
php app/console router:debug

#Debug single route
php app/console router:debug ibw_job_show
